// ========================================================================= // MESIN API V7: TAMBAHAN DATABASE NOMOR WHATSAPP KARYAWAN // ========================================================================= const FOLDER_ID = "12LO7CF2dNTDeE4jEftgC3aoKzZXCtF9X"; function pastikanTabAda(ss, namaTab, headerArray) { let sheet = ss.getSheetByName(namaTab); if (!sheet) { sheet = ss.insertSheet(namaTab); sheet.appendRow(headerArray); sheet.getRange(1, 1, 1, headerArray.length).setFontWeight("bold"); } return sheet; } function doPost(e) { try { let payload = JSON.parse(e.postData.contents); let action = payload.action; let ss = SpreadsheetApp.getActiveSpreadsheet(); pastikanTabAda(ss, "Kasbon", ["Waktu", "Nama", "Nominal Kasbon", "Keterangan"]); pastikanTabAda(ss, "Pengumuman", ["Waktu", "Isi Pengumuman", "Status"]); if (action === "simpan_absen") { let data = payload.data; let folder = DriveApp.getFolderById(FOLDER_ID); let fotoUrl = "-"; if (data.fotoBase64) { let contentType = data.fotoBase64.substring(5, data.fotoBase64.indexOf(';')); let bytes = Utilities.base64Decode(data.fotoBase64.split(',')[1]); let namaFile = data.status.includes("IZIN") || data.status.includes("SAKIT") ? "Surat_" : "Absen_"; let blob = Utilities.newBlob(bytes, contentType, namaFile + data.nama + "_" + new Date().getTime() + ".jpg"); let file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); fotoUrl = "https://drive.google.com/thumbnail?id=" + file.getId() + "&sz=w800"; } let mapsLink = data.lat && data.lng ? `http://googleusercontent.com/maps.google.com/?q=${data.lat},${data.lng}` : "-"; let catatan = data.catatan ? data.catatan : "-"; let sheetAbsen = ss.getSheetByName("Riwayat"); if (sheetAbsen.getRange("I1").getValue() === "") { sheetAbsen.getRange("I1").setValue("Laporan / Keterangan").setFontWeight("bold"); } sheetAbsen.appendRow([data.waktu, data.nama, data.tipe, data.status, data.jarak, mapsLink, fotoUrl, data.perangkat, catatan]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } if (action === "verifikasi_karyawan") { let sheetKaryawan = ss.getSheetByName("Karyawan"); let dataKaryawan = sheetKaryawan.getDataRange().getValues(); for (let i = 1; i < dataKaryawan.length; i++) { if (dataKaryawan[i][0].toString().toUpperCase() === payload.nama.toUpperCase() && dataKaryawan[i][2].toString() === payload.password.toString()) { return ContentService.createTextOutput(JSON.stringify({ status: "success", tipe: dataKaryawan[i][1] })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Password salah!" })).setMimeType(ContentService.MimeType.JSON); } // UPDATE: SIMPAN NOMOR WA DI KOLOM KE-5 (KOLOM E) if (action === "tambah_karyawan") { let sheetKaryawan = ss.getSheetByName("Karyawan"); sheetKaryawan.appendRow([payload.nama.toUpperCase(), payload.tipe, payload.password, payload.gaji, payload.wa || "-"]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } // UPDATE: EDIT NOMOR WA if (action === "edit_karyawan") { let sheetKaryawan = ss.getSheetByName("Karyawan"); let dataKaryawan = sheetKaryawan.getDataRange().getValues(); for (let i = 1; i < dataKaryawan.length; i++) { if (dataKaryawan[i][0].toString().toUpperCase() === payload.nama.toUpperCase()) { sheetKaryawan.getRange(i + 1, 2).setValue(payload.tipe); sheetKaryawan.getRange(i + 1, 3).setValue(payload.password); sheetKaryawan.getRange(i + 1, 4).setValue(payload.gaji); sheetKaryawan.getRange(i + 1, 5).setValue(payload.wa || "-"); // Update kolom ke-5 return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error" })).setMimeType(ContentService.MimeType.JSON); } if (action === "hapus_karyawan") { let sheetKaryawan = ss.getSheetByName("Karyawan"); let dataKaryawan = sheetKaryawan.getDataRange().getValues(); for (let i = 1; i < dataKaryawan.length; i++) { if (dataKaryawan[i][0].toString().toUpperCase() === payload.nama.toUpperCase()) { sheetKaryawan.deleteRow(i + 1); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error" })).setMimeType(ContentService.MimeType.JSON); } if (action === "tambah_kasbon") { let sheetKasbon = ss.getSheetByName("Kasbon"); sheetKasbon.appendRow([payload.waktu, payload.nama.toUpperCase(), payload.nominal, payload.keterangan]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } if (action === "simpan_config_hrd") { let sheetConfig = ss.getSheetByName("ConfigHRD"); if (!sheetConfig) { sheetConfig = ss.insertSheet("ConfigHRD"); } sheetConfig.getRange("A1").setValue("TarifLembur"); sheetConfig.getRange("B1").setValue(payload.tarifLembur); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { let action = e.parameter.action; let ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === "get_riwayat") { let sheet = ss.getSheetByName("Riwayat"); let data = sheet.getDataRange().getValues(); let riwayat = []; for (let i = 1; i < data.length; i++) { riwayat.push({ waktu: data[i][0], nama: data[i][1], tipe: data[i][2], status: data[i][3], jarak: data[i][4], maps: data[i][5], foto: data[i][6], perangkat: data[i][7], catatan: data[i][8] || "-" }); } riwayat.reverse(); return ContentService.createTextOutput(JSON.stringify({ status: "success", data: riwayat })).setMimeType(ContentService.MimeType.JSON); } // UPDATE: AMBIL DATA NOMOR WA if (action === "get_karyawan") { let sheet = ss.getSheetByName("Karyawan"); let data = sheet.getDataRange().getValues(); let listKaryawan = []; for (let i = 1; i < data.length; i++) { listKaryawan.push({ nama: data[i][0], tipe: data[i][1], password: data[i][2], gaji: data[i][3] || 0, wa: data[i][4] || "-" }); } return ContentService.createTextOutput(JSON.stringify({ status: "success", data: listKaryawan })).setMimeType(ContentService.MimeType.JSON); } if (action === "get_kasbon") { let sheet = ss.getSheetByName("Kasbon"); if(!sheet) return ContentService.createTextOutput(JSON.stringify({ status: "success", data: [] })).setMimeType(ContentService.MimeType.JSON); let data = sheet.getDataRange().getValues(); let listKasbon = []; for (let i = 1; i < data.length; i++) { listKasbon.push({ waktu: data[i][0], nama: data[i][1], nominal: data[i][2], keterangan: data[i][3] }); } return ContentService.createTextOutput(JSON.stringify({ status: "success", data: listKasbon })).setMimeType(ContentService.MimeType.JSON); } if (action === "get_config_hrd") { let sheetConfig = ss.getSheetByName("ConfigHRD"); let tarif = 8000; if (sheetConfig) { let val = sheetConfig.getRange("B1").getValue(); if (val) tarif = parseInt(val); } return ContentService.createTextOutput(JSON.stringify({ status: "success", data: { tarifLembur: tarif } })).setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } }
